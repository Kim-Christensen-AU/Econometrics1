\section{Introduction}

\sectionframe{1}{Introduction}

\begin{frame}{Notation}
A \textbf{panel data set} is a combination of cross-sectional and time-series data. It is denoted
$$(y_{it}, \vx_{it}')',$$
for $i=1, \dots,N$ and $t=1,\ldots,T$.

\begin{itemize}
\item Basically, panel data are repeated observations of cross-sectional (individual) units $i$ that are collected over time $t$. The cross-sectional units can be individuals, firms, countries, ...
\item The idea is to model the variable of interest $y_{it}$ as a function of the covariates (regressors) $\vx_{it}$
\item In this lecture, we study \textbf{static} linear panel data models, so $y_{it}$ is a linear function of $\vx_{it}$ (and there are no lagged values of $y_{it}$ in $\vx_{it}$)
\end{itemize}
\end{frame}

\begin{frame}{Data structure}
\centering
\renewcommand{\arraystretch}{1.35}
\setlength{\tabcolsep}{0.65em}
\resizebox{0.95\textwidth}{!}{%
\begin{tabular}{c|cccc}
 & \multicolumn{4}{c}{Time} \\
{Cross-section} & $t=1$ & $t=2$ & $\cdots$ & $t=T$ \\
\hline
$i=1$ & $(y_{11},\vx_{11}')$ & $(y_{12},\vx_{12}')$ & $\cdots$ & $(y_{1T},\vx_{1T}')$ \\
$i=2$ & $(y_{21},\vx_{21}')$ & $(y_{22},\vx_{22}')$ & $\cdots$ & $(y_{2T},\vx_{2T}')$ \\
$\vdots$ & $\vdots$ & $\vdots$ & $\ddots$ & $\vdots$ \\
$i=N$ & $(y_{N1},\vx_{N1}')$ & $(y_{N2},\vx_{N2}')$ & $\cdots$ & $(y_{NT},\vx_{NT}')$
\end{tabular}
}

\vspace{0.4cm}
\begin{itemize}
\item Cell $(i,t)$ has the outcome and regressors for unit $i$ at time $t$.
\item There are no missing data (a balanced panel)
\end{itemize}
\end{frame}

\begin{frame}{Examples of Panel Data}
\begin{enumerate}
\item[$\Rightarrow$] Examples 
\begin{itemize}
\item The Panel Study of Income Dynamics (PSID): a survey containing a large set of individuals with main focus on economic and demographic issues of families
\item The National Longitudinal Surveys (NLS): collects information on labor market activities of several groups of men and women
\item The Penn World Table (PWT): collects purchasing power parity and national income accounts for numerous countries
\item The Danish National Centre for Register-based Research (based at our department)
\item ...
\end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{Why should we use panel data?}
\begin{enumerate}
\item[$\Rightarrow$] Requires/allows modeling unobserved and time-invariant characteristics/heterogeneity of the cross sectional units $i$
\begin{itemize}
\item Some of these variables are measurable, but others are difficult to approximate
\end{itemize}
\item[$\Rightarrow$] Panel data are more informative - larger data sets
\begin{itemize}
\item Allows identifying and measuring effects that are simply not estimable in pure time-series or pure cross-sectional data
\end{itemize}
\item[$\Rightarrow$] Examples:
\begin{itemize}
\item Estimate whether an average increase in a group of individuals can be attributed to a certain subgroup or subgroups
\item Program evaluation, where effectiveness of a training program is evaluated in the groups of participants
\end{itemize}
\item[$\Rightarrow$] However, there are certain limitations
\begin{itemize}
\item Complications related to design, data collection, unclear questions in surveys, etc.
\end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{The linear panel data model}
The linear panel data model is given by
$$
y_{it} = \alpha_i+\vx_{it}'\vbeta+u_{it}, \quad
u_{it} \sim \text{i.i.d.}(0,\sigma_u^2)
$$
for $i=1,\ldots,N$ and $t=1,\ldots,T$.
\begin{itemize}
\item $\vx_{it}$: $K\times 1$ vector of explanatory variables (typically includes a constant, but not always)
\item $\vbeta$: $K\times 1$ vector of slope coefficients
\item $\alpha_i$: unobserved heterogeneity
\item $u_{it}$: error term (independent of $\alpha_i$ and $\vx_{it}$ for all $i$ and  $t$)
\end{itemize}
The random variable $\alpha_{i}$ accounts for any unobserved heterogeneity that is not included in the regression, but that is time-invariant (so it belongs to unit $i$ and it does not change across $t$)
\end{frame}

\section{Pooled OLS}

\sectionframe{2}{The Pooled OLS estimator}


\begin{frame}{Pooled OLS}
Consider the panel model
\[
y_{it} = \alpha_i + \vx_{it}'\vbeta + u_{it},
\qquad i=1,\dots,N,\quad t=1,\dots,T.
\]
\begin{itemize}
\item Since $\alpha_i$ is not observed, an important modeling decision is how to deal with it?

\item A naive approach is to absorb $\alpha_{i}$ into a ``composite'' error term and apply OLS to estimate $\vbeta$ from the regression
$$
y_{it} = \vx_{it}'\vbeta + \varepsilon_{it}
$$
where $\varepsilon_{it} = \alpha_i + u_{it}$

\item This is called Pooled OLS: It ignores the individual effects structure and views all observations as comprising one large cross-section.
\end{itemize}
\end{frame}

\begin{frame}{The Pooled OLS estimator}
\begin{itemize}
\item Stack observations as
\[
\vy =
\begin{pmatrix}
y_{11}\\
\vdots\\
y_{NT}
\end{pmatrix},
\qquad
\mX =
\begin{pmatrix}
\vx_{11}'\\
\vdots\\
\vx_{NT}'
\end{pmatrix}.
\]

\item The pooled OLS estimator is
\begin{align*}
\hat{ \vbeta}_{\mathrm{P}}
&=
(\mX'\mX)^{-1}\mX'\vy \\
&=
\left(
\sum_{i=1}^N \sum_{t=1}^T \vx_{it}\vx_{it}'
\right)^{-1}
\left(
\sum_{i=1}^N \sum_{t=1}^T \vx_{it} \vy_{it}
\right).
\end{align*}

\item $\hat\vbeta_{\mathrm{P}}$ is consistent if $\alpha_i$ is uncorrelated with $\vx_{it}$, i.e.\ $\E[\vx_{it} \alpha_{i}] = 0$
\item However, if $\alpha_i$ and $\vx_{it}$ are uncorrelated, we can construct a more efficient \textbf{random effects} estimator, which exploits that the variance of $\varepsilon_{it}$ is the sum of two unobserved components
$$
\text{var}(\varepsilon_{it}) = \sigma_{ \alpha}^{2} + \sigma_{u}^{2}, \quad \sigma_{ \alpha}^{2} = \text{var}( \alpha_i)
$$
\end{itemize}

\end{frame}

\begin{frame}{Running Example: Job Training and Firm Productivity}
\begin{enumerate}
\item[$\Rightarrow$] We will use this example several times
\item[$\Rightarrow$] How effective are specifically designed training programs in terms of reducing scrap rates

\end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{When Individual Effects Are Correlated With Regressors}
\begin{enumerate}
\item[$\Rightarrow$] Examples of how $\vx_{it}$ may be correlated with $\alpha_i$
\item[$\Rightarrow$] Individual earnings equation:
\begin{itemize}
\item $y_{it} = \log ( \text{wage}_{it})$
\item $\vx_{it}$ contains education level of individual~$i$
\item $\alpha_i$ includes unobserved ability, which is likely to affect~$\vx_{it}$
\end{itemize}
\item[$\Rightarrow$] Firm level investment equation:
\begin{itemize}
\item $y_{it} = \log ( \text{investment}_{it})$
\item $\vx_{it}$ contains cost of capital of firm~$i$
\item $\alpha_i$ includes unobserved managerial quality, which is likely to affect~$\vx_{it}$
\end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{Beyond Pooled OLS}
There are more general (and better) panel data modeling approaches. Which one we should choose depends on the interaction of $\alpha_i$ with $\vx_{it}$

\textbf{Fixed effects (FE) model}:
\begin{itemize}
\item The $\alpha_i$'s are viewed as fixed unknown parameters
\item It allows $E[x_{it} \alpha_i] \neq 0$
\item However, there is a large number of parameters to estimate if the cross-sectional dimension, $N$, is big
\end{itemize}
\textbf{Random Effects (RE) model}: 
\begin{itemize}
\item $\alpha_i$ is assumed to be random, such that $\alpha_i \sim \text{i.i.d.}(0,\sigma_{\alpha}^2)$ and $ E[x_{it}\alpha_i] = 0$ for each~$i$ and~$t$.
\item So instead of $N$ parameters, there is only one ($\sigma_\alpha^2$) to estimate
\item This leads to efficient estimates of $\vbeta$ via feasible Generalized Least Squares (FGLS).
\end{itemize}
\end{frame}

\section{Fixed effects (FE) model}
%%%%%%%%%%%%%%%%%

\sectionframe{2}{Fixed Effects Model}

\begin{frame}{Setup}
The FE model is a linear regression, where the intercept varies across individual units:
$$
y_{it} = \alpha_i + \vx_{it}'\vbeta + u_{it}, \quad u_{it}\sim\text{i.i.d.}(0,\sigma_u^2)
$$
\begin{itemize}
\item The unobserved heterogeneity is regarded as ``parameters''.
\item Brute force estimation of $\alpha_{i}$ and $\vbeta$ is not attractive when $N$ is large, because it requires estimation of many parameters.
\item This also leads to an incidental parameters problem (IPP).
\item Alternatively, there are transformations of the model that eliminate $\alpha_i$ from the regression equation. The drawback of these approaches is that we cannot estimate coefficients on time-invariant regressors with $\vx_{it} = \bar\vx_i$ (e.g., observed heterogeneity)
\end{itemize}
\end{frame}

\begin{frame}{The Within Transformation}
\begin{itemize}
\item Calculate deviations from individual unit means:
$$
y_{it}-\bar{y}_i = (\vx_{it}-\bar\vx_i)' \vbeta + (u_{it}-\bar{u}_i )
$$
where
$$
\frac{1}{T}\sum_{t=1}^T y_{it} = \alpha_i + \frac{1}{T}\sum_{t=1}^T \vx_{it}'\vbeta + \frac{1}{T}\sum_{t=1}^T u_{it}
\quad \Rightarrow \quad \bar{y}_i = 
\alpha_i + \bar\vx_i' \vbeta + \bar{u}_i
$$
\vspace*{-\baselineskip}
\item The FE or ``within'' estimator is the OLS estimator:
$$
\hat\vbeta_{\text{FE}} = \left(\sum_{i=1}^N \sum_{t=1}^T (\vx_{it}-\bar\vx_i ) (\vx_{it}-\bar\vx_i )' \right)^{-1} \sum_{i=1}^N \sum_{t=1}^T (\vx_{it}-\bar\vx_i )(y_{it}-\bar{y}_i )
$$

\item The $N$ intercepts are estimated as $\hat\alpha_i = \bar{y}_i - \bar\vx_i' \hat\vbeta_{\text{FE}},i=1,\ldots ,N$.
\end{itemize}

\end{frame}

\begin{frame}{Large sample properties of $\hat{ \vbeta}_{\text{FE}}$}

\begin{itemize}
\item $\hat\vbeta_{\text{FE}}$ is consistent
$$
\plim_{N \rightarrow \infty}( \hat\vbeta_{\text{FE}}) = \vbeta
$$
\vspace*{-\baselineskip}
\item $T$ is fixed.


\item Assume that the cross-sectional units are i.i.d. Then, under standard regularity conditions, the CLT implies that:
\[
\frac{1}{\sqrt N}
\sum_{i=1}^N
\left(
\sum_{t=1}^T
(\vx_{it}-\bar\vx_i ) (u_{it}-\bar{u}_i )\right)
\overset{d}{\longrightarrow}
N \left(0, \sigma_{u}^{2} Q \right),
\]
where
$
Q = E \left[ \sum_{t=1}^T (\vx_{it}-\bar\vx_i ) (\vx_{it}-\bar\vx_i )'
\right]$.
\end{itemize}

$\Rightarrow$

\[
\sqrt N(\hat\vbeta_{FE}-\vbeta)
\overset{d}{\longrightarrow}
N(0, \sigma_{u}^{2} Q^{-1}),
\]

\end{frame}

\begin{frame}{Covariance matrix estimator}
    
We estimate the covariance matrix of $\hat\vbeta_{\text{FE}}$ using
$$
\widehat\var[\hat\vbeta_{\text{FE}}] = \hat\sigma_u^2 \left( \sum_{i=1}^N \sum_{t=1}^T (\vx_{it} - \bar\vx_i ) (\vx_{it} - \bar\vx_i )' \right)^{-1}
$$
where
\[
\hat\sigma_u^2 = \frac{1}{N(T-1)} \sum_{i=1}^N \sum_{t=1}^T \hat{u}_{it}^2
\qquad \text{and} \qquad \hat{u}_{it} = y_{it} - \hat\alpha_i - \vx_{it}' \hat\vbeta_{\text{FE}}
\]

\end{frame}

\begin{frame}{Least Squares Dummy Variables Estimator}
\begin{itemize}
\item An equivalent approach is to conduct a regression with dummy variables based on unit membership.
\item We define $\valpha = (\alpha_1,\ldots,\alpha_N )'$ and let $\vd_i$ be a vector of size $N$ with 1 as the $i$-th element and zeros elsewhere 

$\Rightarrow$
$$
y_{it} = \vd_i'\valpha + \vx_{it}' \vbeta + u_{it}
$$
\vspace*{-\baselineskip}
\item We estimate the parameters $\valpha$ and $\vbeta$ by OLS. The least squares dummy variables (LSDV) estimator of $\vbeta$ is denoted $\vbeta_{\mathrm{LSDV}}$.
\item $\vbeta_{\mathrm{LSDV}} = \vbeta_{\mathrm{FE}}$ (by the Frisch–Waugh–Lovell theorem).
\item Numerically less appealing, as we need to include many regressors for large $N$ (and end up handling very big matrices).
\item \textbf{Warning}: To avoid multicollinearity either leave the constant out of $\vx_{it}$ or one of the dummies. 
\end{itemize}
\end{frame}

\begin{frame}{Strict Exogeneity}
\begin{itemize}
\item The FE estimator requires the \textbf{strict exogeneity} condition:
$$
\E[\vx_{it} u_{is}] = 0 \text{ for all~$s$ and~$t$}
$$
\item This assumption excludes, in particular, any dependence between $\vx_{it}$ and past values of $y_{it}$
\end{itemize}
\end{frame}

\begin{frame}{The Between Estimator}
\begin{itemize}
\item The ``between'' estimator is obtained by running an OLS regression on the time averages, i.e.
$$
\bar{y}_i = \bar\vx_i' \vbeta + \underbrace{\alpha_i + \bar{u}_i}_{\bar\varepsilon_i}
$$
\vspace*{-\baselineskip}
\item[$\Rightarrow$] The ``between'' estimator is
$$
\hat\vbeta_{\text{B}} = \left(\sum_{i=1}^N (\bar\vx_i - \bar\vx ) (\bar\vx_i - \bar\vx )' \right)^{-1} \sum_{i=1}^N (\bar\vx_i - \bar\vx )(\bar{y}_i - \bar{y} )
$$
\item Consistency assumes strict exogeneity, but it also requires that $\alpha_i$ and $\vx_{it}$ are uncorrelated (a strong assumption)
\item The last condition is not needed for the FE estimator
\item In fact, the ``between'' estimator is really only useful in the context of the RE model
\end{itemize}
\end{frame}

\begin{frame}{The First-Difference Estimator: I}

The ``within'' (or ``between'') transformation is not the only way to eliminate $\alpha_i$ from the FE model.

\begin{itemize}
\item Write the regression equation
\[y_{it} = \alpha_i + \vx_{it}' \vbeta + u_{it}
\]
in terms of first differences:
$$
\Delta y_{it} = \Delta \vx_{it}' \vbeta + \Delta u_{it}
$$
$\Rightarrow$
\[
\hat{\vbeta}_{FD}
=
\left(
\sum_{i=1}^{N}\sum_{t=2}^{T}
\Delta x_{it}\Delta x_{it}'
\right)^{-1}
\left(
\sum_{i=1}^{N}\sum_{t=2}^{T}
\Delta x_{it}\Delta y_{it}
\right),
\]

\item There are only $T-1$ observations in time dimension, since we lose a data point by differencing

\item Normally, there is no intercept in the first-difference regression

\item Consistency of $\hat{\vbeta}_{FD}$ requires \textbf{weak exogeneity}:
$$
E[\Delta \vx_{it} \Delta  u_{it}] = 0 \text{ for all $t$}
$$

\end{itemize}

\end{frame}

\begin{frame}{The First-Difference Estimator: II}

\begin{itemize}

\item The asymptotic distribution of $\hat\vbeta_{FD}$:

\[
\sqrt{N}\left(\hat\vbeta_{FD}-\vbeta\right)
\overset{d}{\to}
\mathcal{N}\left(0,Q^{-1}\Omega Q^{-1}\right),
\]
where
\[
Q
=
E\left[
\sum_{t=2}^T
\Delta\vx_{it}\Delta\vx_{it}'
\right],
\qquad
\Omega
=
\var\left[
\sum_{t=2}^T
\Delta\vx_{it}\Delta u_{it}
\right].
\]

\item Standard errors of $\hat\vbeta_{\text{FD}}$ need to take into account that even if $u_{it}\sim i.i.d.(0,\sigma_u^2)$, differencing creates negative serial correlation in  $\Delta u_{it}$, since $\var[\Delta u_{it}] = 2\sigma_u^2$ and $\cov[\Delta u_{it},\Delta u_{i,t-1}] = -\sigma_u^2$ (while $
\cov[\Delta u_{it},\Delta u_{is}] = 0 \quad\text{if } |t-s|>1$).

\item Thus, in general, $\Omega \neq \sigma_u^2 Q$, and $Q^{-1}\Omega Q^{-1} \neq \sigma_u^2 Q^{-1}$. Therefore, the usual i.i.d. OLS variance formula is incorrect, and we use the sandwich estimator (clustered at the individual level).

\item As for the FE estimator, the FD approach cannot estimate coefficients on regressors that are time-invariant ($\Delta\vx_{it}=0$)

\end{itemize}

\end{frame}

\section{Random effects (RE) model}
%%%%%%%%%%%%%%%%%

\sectionframe{3}{Random Effects Model}

\begin{frame}{Setup}

In the RE model, we explicitly model $\alpha_{i}$ as a random variable (and, hence, include as part of the error term)

\begin{itemize}
\item The RE model is
\begin{align*}
y_{it} &= \beta_0 + \vx_{it}' \vbeta + \alpha_i + u_{it}\\
u_{it} &\sim\text{i.i.d.}(0,\sigma_u^2)\\
\alpha_i &\sim\text{i.i.d.}(0,\sigma_{\alpha}^2)
\end{align*}
\vspace*{-\baselineskip}
\begin{itemize}
\item $\alpha_i$ and $u_{it}$ are mutually independent
\item $\alpha_i$ and $u_{it}$ are independent from $\vx_{js}$ for each~$j$ and~$s$ (even stronger than strict exogeneity)
\end{itemize}
\item[$\Rightarrow$] (Pooled) OLS estimator of $\beta_0, \vbeta$ is consistent but:
\begin{itemize}
\item The usual standard errors are wrong (serial correlation in errors induced by $\alpha_i$) 
\item OLS is not efficient as it does not take advantage of the particular form of the error covariance matrix 
\end{itemize}
\item[$\Rightarrow$] For the price of the exogeneity condition we can obtain more efficient estimator
\end{itemize}
\end{frame}

\begin{frame}{Random Effects: GLS Intuition}
\begin{enumerate}
\item[$\Rightarrow$] Hence we develop the {\bf Generalized Least Squares (GLS)} estimator that explores the particular error structure
\item[$\Rightarrow$] Intuition behind GLS: apply OLS to a transformed data such that covariance matrix of the transformed error is an identity matrix
\item[$\Rightarrow$] Define $\vu_i = (u_{i1},\ldots,u_{iT})'$, $\viota_T = (1,1,\ldots,1)'$, and the $T\times T$ identity matrix~$\mI_T$
\item[$\Rightarrow$] The composite error term is then $\vvarepsilon_i = \alpha_i \viota_T +\vu_i$
\item[$\Rightarrow$] Write model as (ignoring constant term or including it in~$\vx_i$)
$$
\underbrace{\vy_i}_{T\times 1} = \underbrace{\mX_i'}_{T\times K}\underbrace{\vbeta}_{K\times 1}+\underbrace{\vvarepsilon_i}_{T\times 1}
\quad \text{with} \quad \E[\vvarepsilon_i] = \vzeros ,
\quad \var[\vvarepsilon_i] = \underbrace{\mOmega}_{T\times T}
$$
where $\vy_i = (y_{i1},\ldots,y_{iT})'$ and $\mX_i = (\vx_{i1},\ldots,\vx_{iT})$
\item[$\Rightarrow$] If we knew the matrix $\mOmega$ then OLS on the transformed model
$$
\mOmega^{-1/2}\vy_i = \mOmega^{-1/2}\mX_i'\vbeta + \mOmega^{-1/2}\vvarepsilon_i 
$$
gives the GLS estimator
\end{enumerate}
\end{frame}

\begin{frame}{The GLS Estimator}
\begin{enumerate}
\item[$\Rightarrow$] If $\mOmega$ was known, then the (efficient) GLS estimator can be computed immediately:
$$
\hat\vbeta_{\text{GLS}} = \left(\sum_{i=1}^N \mX_i \mOmega^{-1} \mX_i'\right)^{-1} \sum_{i=1}^N \mX_i \mOmega^{-1} \vy_i 
$$
\item[$\Rightarrow$] Assuming the law of $\vvarepsilon_i$ is normal (conditional on $\mX$):
$$
\hat\vbeta_{\text{GLS}} \sim \mathcal{N} \left( \vbeta , \left(\sum_{i=1}^N \mX_i \mOmega^{-1} \mX_i' \right)^{-1}\right)
$$
\item[$\Rightarrow$] As usual, this also holds without normality, but then only asymptotically 
\item[$\Rightarrow$] Follows from standard linear manipulations and by noting that $\E[\sum_{i=1}^N \mX_i \mOmega^{-1} \vvarepsilon_i \vvarepsilon_i' \mOmega^{-1} \mX_i' |\mX] =\sum_{i=1}^N \mX_i \mOmega^{-1} \mX_i'$
\end{enumerate}
\end{frame}

\begin{frame}{Error Covariance Matrix}
\begin{align*}
\mOmega = \var[\vvarepsilon_i] 
&= \var[\alpha_i \viota_T + \vu_i] \\
&= \sigma_{\alpha}^2\viota_T\viota_T' + \sigma_u^2\mI_T \\
&= \left(
\begin{array}{cccccc}
\sigma^2_\alpha+\sigma_u^2&\sigma_\alpha^2&\dots&\sigma_\alpha^2\\
\sigma_\alpha^2&\sigma^2_\alpha+\sigma_u^2&\dots&\sigma_\alpha^2\\
\vdots&\vdots&\ddots&\vdots\\
\sigma_\alpha^2&\sigma_\alpha^2&\dots&\sigma^2_\alpha+\sigma_u^2
\end{array}
\right)
\end{align*}
\begin{itemize}
\item[$\Rightarrow$] This matrix has $\sigma^2_\alpha+\sigma_u^2$ on the diagonal and $\sigma_\alpha^2$ elsewhere
\item[$\Rightarrow$] Of course, $\sigma^2_\alpha$ and $\sigma_u^2$ are unknown in practice
\end{itemize}
\end{frame}

\begin{frame}{Quasi-Demeaning}
\begin{enumerate}
\item[$\Rightarrow$] With a ton of algebra, it is possible to show that 
\begin{align*}
\hat\vbeta_{\text{GLS}} ={}& \left( \sum_{i=1}^N \sum_{t=1}^T (\vx_{it} -\bar\vx_i )(\vx_{it} -\bar\vx_i )' + \psi T \sum_{i=1}^N (\bar\vx_i - \bar\vx)(\bar\vx_i - \bar\vx)' \right)^{-1} \\
&\times \left( \sum_{i=1}^N \sum_{t=1}^T (\vx_{it} -\bar\vx_i )(y_{it} -\bar{y}_i )' + \psi T \sum_{i=1}^N (\bar\vx_i - \bar\vx)(\bar{y}_i - \bar{y})' \right) ,
\end{align*}
where
$$
\psi = \frac{\sigma_u^2}{\sigma_u^2 + T \sigma_\alpha^2}
$$
\item[$\Rightarrow$] This is merely an alternative derivation of $\hat\vbeta_{\text{GLS}}$. However, it has a possible computational advantage, since we do not invert a big $T \times T$ matrix.
\item[$\Rightarrow$] We need to estimate $\sigma^2_\alpha$ and $\sigma_u^2$ for a feasible GLS estimator.
\end{enumerate}
\end{frame}

\begin{frame}{Feasible GLS estimator}
\begin{enumerate}
\item[$\Rightarrow$] Based on the FE (within) estimator, we can estimate $\sigma_u^2$:
\begin{align*}
\hat{u}_{it} &= y_{it} - \hat\alpha_i - \vx_{it}'\hat\vbeta_{\text{FE}} \\
\hat\sigma_u^2 &= \frac{1}{N(T-1)}\sum_{i=1}^N \sum_{t=1}^T \hat{u}_{it}^2
\end{align*}
\item[$\Rightarrow$] Based on the ``between'' estimator, we can estimate $\sigma_\alpha^2$:
\begin{itemize}
\item Recall that the between model is based on $\bar{y}_i = \beta_0 + \bar\vx_i' \vbeta +\bar\varepsilon_i$. Hence, the variance of $\bar\varepsilon_i$ is $\var[\bar\varepsilon_i] = \var[\alpha_i + \bar{u}_i] = \sigma_\alpha^2+T^{-1}\sigma_u^2$
\item Then
$$
\frac{1}{N}\sum_{i=1}^N (\bar{y}_i - \hat\beta_{0,\text{B}} - \bar\vx_i' \hat\vbeta_{\text{B}} )^2 
= \hat\sigma_\alpha^2 + \frac{1}{T} \hat\sigma_u^2
$$
\item Plug-in $\hat\sigma_u^2$ from above and solve for $\hat\sigma_\alpha^2$
\end{itemize}
\item[$\Rightarrow$] The feasible GLS estimator based on the above estimates is known as the \textbf{random effects (RE) estimator}, $\hat\vbeta_{\text{RE}}$.
\end{enumerate}
\end{frame}

\section{FE or RE? The Hausman test}
%%%%%%%%%%%%%%%%%

\sectionframe{5}{FE vs. RE Models: Hausman Test}

\begin{frame}{FE vs.\ RE: Hausman Test Intuition}
\begin{enumerate}
\item[$\Rightarrow$] It is hard to choose between the FE and RE estimators
\item[$\Rightarrow$] One criterion can be built upon whether $\alpha_i$ and $\vx_{it}$ are uncorrelated
\item[$\Rightarrow$] Hausman test is built on this logic 
\item[$\Rightarrow$] General idea of a Hausman test:
\begin{itemize}
\item Suppose you have two estimators, $\hat\theta_A$ and $\hat\theta_B$
\item Under one hypothesis, both $\hat\theta_A$ and $\hat\theta_B$ are consistent, but $\hat\theta_A$ is more efficient than $\hat\theta_B$
\item Under the other hypothesis, $\hat\theta_A$ is inconsistent and $\hat\theta_B$ is consistent
\item Intuition: $\hat\theta_A$ is efficient under ideal circumstances, but not robust, and $\hat\theta_B$ is robust, but not efficient
\end{itemize}
\item[$\Rightarrow$] In the present setup, suppose $\E[u_{it} \vx_{is}] =0$ for all $t$ and $s$, so that:
\begin{itemize}
\item If $\alpha_i$ and $\vx_{it}$ are uncorrelated then both $\hat\vbeta_{\text{FE}}$ and $\hat\vbeta_{\text{RE}}$ are consistent, but $\hat\vbeta_{\text{RE}}$ is more efficient
\item If $\alpha_i$ and $\vx_{it}$ are correlated then $\hat\vbeta_{\text{RE}}$ is inconsistent while $\hat\vbeta_{\text{FE}}$ is still consistent
\end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}{FE vs.\ RE: Hausman Test Statistic}
\begin{enumerate}
\item[$\Rightarrow$] Hausman test formally...
\item[$\Rightarrow$] Under the null of the test, $\alpha_i$ and $\vx_{it}$ are uncorrelated
\item[$\Rightarrow$] This implies that both $\hat\vbeta_{\text{RE}}$ and $\hat\vbeta_{\text{FE}}$ are consistent, hence the distance $\hat\vbeta_{\text{FE}}-\hat\vbeta_{\text{RE}}$ must not be significantly different from zero
\item[$\Rightarrow$] If $\hat\vbeta_{\text{FE}}-\hat\vbeta_{\text{RE}}$ is statistically significantly different from zero, then $\alpha_i$ and $\vx_{it}$ are correlated
\item[$\Rightarrow$] Test statistic is
$$
\xi_{\text{H}} = (\hat\vbeta_{\text{FE}}-\hat\vbeta_{\text{RE}})'\left(\widehat\var[\hat\vbeta_{\text{FE}}]-\widehat\var[\hat\vbeta_{\text{RE}}]\right)^{-1}(\hat\vbeta_{\text{FE}}-\hat\vbeta_{\text{RE}})
$$
\item[$\Rightarrow$] Under the null, $\xi_{\text{H}}\sim\chi^2(K)$ asymptotically, where $\chi^2 (K)$ is a chi-squared distribution with $K$ degrees of freedom
\item[$\Rightarrow$] Note that we used the curious result that $\var[\hat\vbeta_{\text{FE}}-\hat\vbeta_{\text{RE}}] = \var[\hat\vbeta_{\text{FE}}]-\var[\hat\vbeta_{\text{RE}}]$ under the null
\item[$\Rightarrow$] If test rejects, then use $\hat\vbeta_{\text{FE}}$, otherwise use $\hat\vbeta_{\text{RE}}$
\end{enumerate}
\end{frame}

\section{Cluster-robust inference}

\sectionframe{5}{Cluster-Robust Inference}

\begin{frame}{Cluster-Robust Standard Errors}
\begin{enumerate}
\item[$\Rightarrow$] Textbook section 10.2.7 discusses cluster-robust inference
\item[$\Rightarrow$] Generalization of White's heteroskedasticity-robust standard errors% (``robust'' in Stata)
\item[$\Rightarrow$] Non-panel case: suppose data can be divided into $G$ clusters
$$
\underbrace{\vy_g}_{N_g \times 1} = \underbrace{\mX_g'}_{N_g \times K} \vbeta + \underbrace{\vu_g}_{N_g \times 1} , \quad g=1, \ldots , G
$$
\vspace*{-0.6\baselineskip}
\begin{itemize}
\item The clusters are independent of each other
\item We make no assumptions about dependence within clusters
\end{itemize}
\item[$\Rightarrow$] The variance of OLS estimator $\hat\vbeta = \vbeta_0 + (\mX' \mX )^{-1} \sum_{g=1}^G \mX_g \vu_g$ is $(\mX' \mX )^{-1} \sum_{g=1}^G \mX_g \mOmega_g \mX_g'  (\mX' \mX )^{-1}$, where $\mOmega_g = \var[\vu_g]$
\item[$\Rightarrow$] The cluster-robust variance estimator (CRVE) is
$$
\scalebox{0.9}{$\widehat\var[\hat\vbeta] = (\mX' \mX )^{-1} \displaystyle\sum_{g=1}^G \mX_g \hat\vu_g \hat\vu_g' \mX_g'  (\mX' \mX )^{-1}$}
$$
\vspace*{-0.6\baselineskip}
\item[$\Rightarrow$] MacKinnon, Nielsen, Webb (2023): ``Cluster-robust inference: A guide to empirical practice'', \emph{Journal of Econometrics}
\end{enumerate}
\end{frame}
